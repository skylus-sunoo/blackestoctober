scripted_gui = {
	
	sky_utilities_enable = {
		context_type = player_context
		
		parent_window_token = politics_tab
		window_name = "sky_utilities_open"

		visible = {
			is_ai = no
		}

		effects = {
			sky_utilities_open_button_click = {
				set_global_flag = sky_enabled
				if = {
					limit = {
						NOT = {
							OR = {
								has_global_flag = sky_container_1
								has_global_flag = sky_container_2
								has_global_flag = sky_container_3
							}
						}
					}
					set_global_flag = sky_container_1
				}
			}
			sky_utilities_close_button_click = {
				clr_global_flag = sky_enabled
			}
		}
		triggers = {
			no_faction_visible = {
				is_in_faction = no
			}
			sky_utilities_open_button_visible = {
				NOT = {
					has_global_flag = sky_enabled
				}
			}
			sky_utilities_close_button_visible = {
				has_global_flag = sky_enabled
			}
		}
	}
}
